Open the dashboard at http://localhost:8080

Then open browser console (F12) and run these commands:

    // Check if Chart is available
    console.log('Chart.js available:', typeof Chart);
    
    // Check if Alpine is available
    console.log('Alpine available:', typeof Alpine);
    
    // Get Alpine component
    const dashboard = Alpine.$data(document.querySelector('[x-data]'));
    console.log('Dashboard data:', dashboard);
    
    // Check selected report
    console.log('Selected report:', dashboard.selectedReport);
    
    // Manually trigger chart update
    dashboard.updateCharts();
    
    // Check questions
    console.log('Questions:', dashboard.questions);
    
    // Test showing modal
    if (dashboard.questions && dashboard.questions.length > 0) {
        dashboard.showQuestionDetail(dashboard.questions[0]);
    }
